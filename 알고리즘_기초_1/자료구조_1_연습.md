# 단어 뒤집기 2 (17413)

```swift
let input = readLine()!
var answer = ""
var temp = ""
var doReverse = true

for char in input {
    if char.isLowercase || char.isNumber {
        temp += String(char)
    } else if char == " " {
        if doReverse {
            answer += String(temp.reversed())
        } else {
            answer += temp
        }
        answer += " "
        temp = ""
    } else if char == "<" {
        if temp != "" {
            answer += String(temp.reversed())
            temp = ""
        }
        temp += String(char)
        doReverse = false
    } else if char == ">" {
        temp += String(char)
        answer += temp
        temp = ""
        doReverse = true
    }
}

if temp != "" {
    answer += String(temp.reversed())
}

print(answer)
```


# 쇠막대기 (10799)

```swift
let input = readLine()!
var stack = [Character]()
var answer = 0
var check = false

for char in input {
    if char == "(" {
        stack.append(char)
        check = true
    } else if char == ")" {
        if check {
            stack.removeLast()
            answer += stack.count
            check = false
        } else {
            stack.removeLast()
            answer += 1
        }
    }
}

print(answer)
```

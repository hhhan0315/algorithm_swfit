## 스택 10828
```swift
let num = Int(readLine()!)!
var stack = Stack()

for _ in 0..<num {
    let orders = readLine()!.split(separator: " ").map{String($0)}
    
    switch orders[0] {
    case "push":
        stack.push(Int(orders[1])!)
    case "pop":
        print(stack.pop())
    case "size":
        print(stack.size())
    case "empty":
        print(stack.empty())
    case "top":
        print(stack.top())
    default: break
    }
}

class Stack {
    var memory = [Int]()
    
    func push(_ x: Int) {
        memory.append(x)
    }
    
    func pop() -> Int {
        if isEmpty() {
            return -1
        } else {
            return memory.removeLast()
        }
    }
    
    func isEmpty() -> Bool {
        if memory.isEmpty {
            return true
        } else {
            return false
        }
    }
    
    func empty() -> Int {
        if isEmpty() {
            return 1
        } else {
            return 0
        }
    }
    
    func size() -> Int {
        return memory.count
    }
    
    func top() -> Int {
        if isEmpty() {
            return -1
        } else {
            return memory.last!
        }
    }
}
```

## 단어 뒤집기 9093
```swift
let t = Int(readLine()!)!

for _ in 0..<t {
    let words = readLine()!.split(separator: " ").map{String($0)}
    var arr = [String]()
    
    for word in words {
        var temp = ""
        for str in word.reversed() {
            temp += String(str)
        }
        arr.append(temp)
    }
    
    let result = arr.joined(separator: " ")
    print(result)
}
```

## 괄호 9012

```swift
let t = Int(readLine()!)!

for _ in 0..<t {
    let input = readLine()!

    if isBracket(input) {
        print("YES")
    } else {
        print("NO")
    }
}

func isBracket(_ input: String) -> Bool {
    var stack = [Character]()
    
    for char in input {
        if char == "(" {
            stack.append("(")
        } else {
            if stack.isEmpty {
                return false
            } else {
                stack.removeLast()
            }
        }
    }
    
    if stack.isEmpty {
        return true
    } else {
        return false
    }
}
```



## 스택 수열 1874

```swift
let n = Int(readLine()!)!
var nums = [Int]()
var stack = [Int]()
var operators = [Character]()
var index = 0

for _ in 0..<n {
    let num = Int(readLine()!)!
    nums.append(num)
}

for i in 1...n {
    stack.append(i)
    operators.append("+")
    
    while !stack.isEmpty && nums[index] == stack.last! {
        stack.removeLast()
        operators.append("-")
        index += 1
    }
}

if stack.isEmpty {
    for op in operators {
        print(op)
    }
} else {
    print("NO")
}
```

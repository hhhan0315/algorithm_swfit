# 후위 표기식2 (1935)

```swift
import Foundation

let n = Int(readLine()!)!
var input = Array(readLine()!)
var start: Character = "A"
var stack = [Double]()

for _ in 0..<n {
    let num = readLine()!
    
    for i in input.indices {
        if input[i] == start {
            input[i] = Character(num)
        }
    }
    
    let value = start.asciiValue! + 1
    start = Character(UnicodeScalar(value))
}

for char in input {
    if char.isNumber {
        let value = Double(String(char))!
        stack.append(value)
    } else {
        let b = stack.removeLast()
        let a = stack.removeLast()
        
        switch char {
        case "+":
            stack.append(a + b)
        case "-":
            stack.append(a - b)
        case "*":
            stack.append(a * b)
        case "/":
            stack.append(a / b)
        default: break
        }
    }
}

let result = String(format: "%.2f", stack.first!)
print(result)


```

# 후위 표기식 (1918)

```swift
let input = Array(readLine()!)
var result = ""
var stack = [Character]()

for c in input {
    if c.isUppercase {
        result += String(c)
    } else {
        switch c {
        case "(":
            stack.append(c)
        case "*", "/":
            while !stack.isEmpty && (stack.last == "*" || stack.last == "/") {
                result += String(stack.removeLast())
            }
            stack.append(c)
        case "+", "-":
            while !stack.isEmpty && stack.last != "(" {
                result += String(stack.removeLast())
            }
            stack.append(c)
        case ")":
            while !stack.isEmpty && stack.last != "(" {
                result += String(stack.removeLast())
            }
            stack.removeLast()
        default: break
        }
    }
}

while !stack.isEmpty {
    result += String(stack.removeLast())
}

print(result)

```
